using System.Diagnostics;

namespace Day4
{
    internal class Program
    {
        static void Main(string[] args)
        {  //Card Number Validation algorithm

            
            Console.WriteLine("Enter a number:");
            string input = Console.ReadLine();//getting input from user
            char[] inputArray = input.ToCharArray();//converting into char array
            int n = input.Length;
            char[] reverseArray = new char[n];
            int totalSum = 0;

            //reversing the char array
            for (int i = 0; i < n; i++)
            {
                reverseArray[i] = inputArray[n - 1 - i];
            }

            //Iterating over reversed array and if the position is odd directly adding to totalSum or else multiply the number with 2 and adding it.
            for (int i = 0; i < n; i++)
            {
                int num = int.Parse(reverseArray[i].ToString());//converting character to string and then parsing to integer
                if (i % 2 == 0)
                {
                    int evenPositionNumber = num * 2;
                    int evenSum = 0;
                    //if the multiplied result is a double digit number,add the sum of digits to total sum
                    if (evenPositionNumber > 9)
                    {
                        while (evenPositionNumber != 0)
                        {
                            evenSum += evenPositionNumber % 10;
                            evenPositionNumber /= 10;
                        }
                    }
                    else
                    {
                        evenSum = evenPositionNumber;
                    }
                    totalSum += evenSum;
                }
                else
                {
                    totalSum += num;
                }
            }
            //if the totalsum %10 equals to 0,result is Valid Card Number
            if (totalSum % 10 == 0)
            {
                Console.WriteLine("Valid Card number");
            }
            else
            {
                Console.WriteLine("Not a Valid Card number");
            }
        }
    }
}
